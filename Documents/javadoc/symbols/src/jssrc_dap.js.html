 12 /**
 13  * This class contains information about Dolby Audio Processing.
 14  * @constructor
 15  */
 16 var DolbyAudioProcessing = function () {
 17 };
 18 
 19 var cordova = require('cordova'),
 20 exec = require('cordova/exec');
 21 
 22 DolbyAudioProcessing.prototype = {
 23 	/**
 24 	 * The first method to initialize Dolby Audio Processing. Internally binds to the background Dolby service.
 25 	 *
 26 	 * NOTES:
 27 	 *
 28 	 * This typically can be called in the Cordova's deviceready event callback function.
 29 	 * Defaults to DS enabled, Game profile upon initialization, unless the initial profile is defined by the profile parameter.
 30 	 *
 31 	 * @param {String} profile Profile is used to initialize Dolby Audio Processing; the Game profile is used by default if it is null or undefined.
 32 	 * @param {Function} success This callback executes when Dolby Audio Processing is initialized successfully.
 33 	 * @param {Function} fail This callback executes when an error occurs when Dolby Audio Processing is initialized. The PluginResult includes the {@link DapError.DAP_UNAVAILABLE} message.
 34 	 * This callback is invoked when Dolby Audio Processing is not available on a device or an internal error occurs inside the Dolby service or Android OS.
 35 	 *
 36 	 * @example
 37 	 *	dolby.dap.initialize(DapProfile.GAME, onSuccess, onFail);
 38 	 *	function onSuccess() {
 39 	 *		alert("Dolby Audio Processing is ready to use.");
 40 	 *	},
 41 	 *	function onFail(err) {
 42 	 *		alert(err.message);
 43 	 *	},
 44 	 *
 45 	 * @see DapProfile
 46 	 * @see DapError
 47 	 */
 48 	initialize: function (profile, success, fail) {
 49 		exec(success, fail, "DolbyDapPlugin", "initialize",[profile]);
 50 	},
 51 	
 52 	/**
 53 	 * Releases all resources, restoring the Dolby Audio Processing system configuration to the state prior to application control.
 54 	 *
 55 	 * After this call, Dolby Audio Processing methods are not accessible.
 56 	 *
 57 	 * We recommend this call in HTML body's onunload event (when the HTML page is closed).
 58 	 * Using release() API makes sure all resources are released when the plug-in is destroyed. This call is optional in your code.
 59 	 *
 60 	 * @example
 61 	 *	dolby.dap.release();
 62 	 */
 63 	release: function () {
 64 		exec(null, null, "DolbyDapPlugin", "release",[]);
 65 	},
 66 	
 67 	/**
 68 	 * Query the enabled state of Dolby Audio Processing.
 69 	 *
 70 	 * @param {Function} success This callback executes when Dolby Audio Processing status is retrieved successfully; the status is included in PluginResult.
 71 	 * @param {Function} fail This callback executes when Dolby Audio Processing failed to get the status, the PluginResult would include the {@link DapError.NOT_CONNECTED} message.
 72 	 * The failure callback is invoked when the initialize() API is not called.
 73 	 *
 74 	 * @example
 75 	 *	dolby.dap.isEnabled(onSuccess, onFail);
 76 	 *	function onSuccess(value) {
 77 	 *		alert("Dolby Audio Processing is enabled:" + value);
 78 	 *	},
 79 	 *
 80 	 * @see DapError
 81 	 */
 82 	isEnabled: function (success, fail) {
 83 		exec(success, fail, "DolbyDapPlugin", "isEnabled",[]);
 84 	},
 85 	
 86 	/**
 87 	 * Enable/disable Dolby Audio Processing.
 88 	 *
 89 	 * @param {Boolean} enable The enable/disable state to set Dolby Audio Processing.
 90 	 * @param {Function} success This callback executes when Dolby Audio Processing status is set successfully
 91 	 * @param {Function} fail This callback executes when Dolby Audio Processing fails to set the status. The PluginResult would include {@link DapError.NOT_CONNECTED} or {@link DapError.ACTION_NOT_ALLOWED} message.
 92 	 * This error callback is invoked if initialize() API is not invoked before the enable/disable call, or run-time internal errors occur.
 93 	 *
 94 	 * @example
 95 	 *	dolby.dap.setEnabled(true,onSuccess, onFail);
 96 	 *	function onSuccess() {
 97 	 *		alert("Dolby Audio Processing is truned on.");
 98 	 *	},
 99 	 *
100 	 * @see DapError
101 	 */
102 	setEnabled: function (enable, success, fail) {
103 		exec(success, fail, "DolbyDapPlugin", "setEnabled",[enable]);
104 	},
105 	
106 	/**
107 	 * Applies the selected Dolby Audio Processing profile.
108 	 *
109 	 * @param {Function} success A callback that executes when a Dolby Audio Processing profile is applied successfully. The profile name is included in PluginResult.
110 	 * @param {Function} fail A callback that executes when Dolby Audio Processing failed to apply the selected profile. The PluginResult would include the {@link DapError.NOT_CONNECTED} message.
111 	 * This error callback is invoked if initialize() API is not called before the background Dolby service connects. 
112 	 *
113 	 * @example
114 	 *	dolby.dap.getSelectedProfile(onSuccess, onFail);
115 	 *	function onSuccess(value) {
116 	 *		alert("The selected profile is:" + value);
117 	 *	},
118 	 *
119 	 * @see DapError
120 	 */
121 	getSelectedProfile: function (success, fail) {
122 		exec(success, fail, "DolbyDapPlugin", "getSelectedProfile",[]);
123 	},
124 	
125 	/**
126 	 *Select a Dolby Audio Processing profile.
127 	 *
128 	 * @param {String} profile Specifies the profile selection.
129 	 * Exception: {@link DapProfile.DOLBY_PRIVATE_PROFILE}. {@link DapProfile.DOLBY_PRIVATE_PROFILE} is reserved for internal or OEM use only.
130 	 * @param {Function} success This callback executes when Dolby Audio Processing profile activates successfully.
131 	 * @param {Function} fail This callback executes when Dolby Audio Processing fails to set the profile. The PluginResult includes an error message, {@link DapError.NOT_CONNECTED}, {@link DapError.ACTION_NOT_ALLOWED}, or {@link DapError.RUNTIME_ERR}.
132 	 *
133 	 * @example
134 	 *	dolby.dap.setProfile(DapProfile.MOVIE, onSuccess, onFail);
135 	 *	function onSuccess() {
136 	 *		alert("The selected profile is changed.");
137 	 *	},
138 	 *
139 	 * @see DapProfile
140 	 * @see DapError
141 	 */
142 	setProfile: function (profile, success, fail) {
143 		exec(success, fail, "DolbyDapPlugin", "setProfile",[profile]);
144 	},
145 	
146 	/**
147 	 * Suspends the application Dolby Audio Processing session.
148 	 *
149 	 * When the application exits the foreground, this suspends the application Dolby Audio Processing session and restores the system-wide configuration.
150 	 * Typically it would be called in the Cordova's pause  event callback function.
151 	 *
152 	 * @example
153 	 *	dolby.dap.suspendSession();
154 	 */
155 	suspendSession: function () {
156 		exec(null, null, "DolbyDapPlugin", "suspendSession",[]);
157 	},
158 	
159 	/**
160 	 * Resumes the application Dolby Audio Processing session.
161 	 *
162 	 * When the application reenters the foreground, this call resumes the application Dolby Audio Processing session that was suspended on foreground exit by suspendSession() API.
163 	 * Typically it would be called in the Cordova's resume event callback function.
164 	 *
165 	 * @param {Function} success This callback executes when the application configuration is resumed successfully. The getSelectedProfile() API can be invoked to get the selected profile.
166 	 * @param {Function} fail A callback that executes when the application configuration failed to resume.  The PluginResult would include the error {@link DapError.NOT_CONNECTED} or {@link DapError.RUNTIME_ERR}.
167 	 *
168 	 * @example
169 	 *	dolby.dap.restartSession(function () {
170 	 *		dolby.dap.getSelectedProfile(function(value) {
171 	 *			alert("proflie: " + value);
172 	 *				}, null);
173 	 *			}, null);
174 	 *
175 	 * @see DapError
176 	 */
177 	restartSession: function (success, fail) {
178 		exec(success, fail, "DolbyDapPlugin", "restartSession",[]);
179 	}
180 };
181 
182 var dap = new DolbyAudioProcessing();
183 module.exports = dap;
