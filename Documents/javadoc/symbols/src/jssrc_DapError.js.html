 12 /**
 13  * This is the Dolby Audio Processing error object.
 14  *
 15  * @constructor
 16  * @param {Number} code Error code
 17  * @param {String} message Error message description
 18  */
 19 var DapError = function (code, message) {
 20 	this.code = code || null;
 21 	this.message = message || '';
 22 };
 23 
 24 /**
 25  * This error returns when Dolby Audio Processing is not available on a device. The error indicates the onFail callback of initialize() API when Dolby Audio Processing is not available on a device.
 26  */
 27 DapError.DAP_UNAVAILABLE = 1;
 28 
 29 /**
 30  * This error returns when the background Dolby service is not connected. The initialize() API should be called if it occurs. If initialize() API is not invoked, the other APIs return this error .
 31  * In an unusual scenario (e.g. Android OS memory is used up), if the background Dolby service is restarted, the error can occur before the initialize() API is called to reinitialize Dolby Audio  Processing.
 32  */
 33 DapError.NOT_CONNECTED = 2;
 34 
 35 /**
 36  * This error returns when when the input parameter is invalid.
 37  */
 38 DapError.INVALID_PARAM = 3;
 39 
 40 /**
 41  * This error returns when the action is not allowed temporarily, e.g. if the user tries to switch the profile when Dolby Audio Processing is turned off.
 42  * We recommend first turning on Dolby Audio Processing, then switching the active profile.
 43  * The dolby.dap.setEnable()/dolby.dap.setProfile() API returns this error code if dolby.dap.suspendSession() API has been called.
 44 
 45 --------------------------------------------------------------------------------
 46 
 47 
 48  */
 49 DapError.ACTION_NOT_ALLOWED = 4;
 50 
 51 /**
 52  * This error returns when a run-time error occurs, e.g. an Android OS or Dolby service internal error. The error indicates that Dolby Audio Processing is not working correctly.
 53  */
 54 DapError.RUNTIME_ERR = 5;
 55 
 56 
 57 module.exports = DapError;
